  /* === Google Sites iframe resize === */
  (function(){
    // Continuously tell parent frame our height so it can resize
    function postHeight(){
      const h = document.documentElement.scrollHeight;
      // Standard postMessage for iframe resize
      window.parent.postMessage({type:'resize',height:h},'*');
      // Google Sites specific
      window.parent.postMessage(JSON.stringify({method:'setHeight',value:h}),'*');
    }
    // Post on load and on resize
    postHeight();
    window.addEventListener('load', postHeight);
    window.addEventListener('resize', postHeight);
    // Keep posting periodically for dynamic content
    setInterval(postHeight, 1000);

    // Also try to break out of small iframe constraints
    try {
      if(window.frameElement){
        window.frameElement.style.width='100%';
        window.frameElement.style.height='100vh';
        window.frameElement.style.minHeight='100vh';
        window.frameElement.style.border='none';
      }
    } catch(e){}
